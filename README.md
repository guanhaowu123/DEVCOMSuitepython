# install DEVCOMSuitepython
```
git clone https://github.com/guanhaowu123/DEVCOMSuitepython.git
cd DEVCOMSuitepython
conda env create -n DEVCOMSuitepy -f DEVCOMSuitepy.yaml
conda activate DEVCOMSuitepy
pip install git+https://github.com/guanhaowu123/DEVCOMSuitepython.git
cd ~
git clone https://github.com/axelalmet/flowsig.git
cd ./flowsig/
pip3 install .

```
